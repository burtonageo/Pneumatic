cmake_minimum_required (VERSION 2.8)

set(PROJECT_NAME glCube)
set(PROJECT_VERSION_MAJOR 0)
set(PROJECT_VERSION_MINOR 1)
project(${PROJECT_NAME})

find_package(GLEW REQUIRED)
find_package(OpenGL REQUIRED)

find_package(PkgConfig REQUIRED)
pkg_search_module(GLFW REQUIRED glfw3)

set(SRC_DIR "${CMAKE_SOURCE_DIRECTORY}src")
set(INCLUDE_DIR "${CMAKE_SOURCE_DIRECTORY}include")
set(SHADER_DIR "${CMAKE_SOURCE_DIRECTORY}shader")
set(SOURCES
	"${SRC_DIR}/Game.cpp"
	"${SRC_DIR}/GLWindow.cpp")

configure_file (
	"${INCLUDE_DIR}/GLWindow.hpp.cmake"
	"${PROJECT_BINARY_DIR}/GLWindow.hpp")

include_directories(
	${SRC_DIR}
	${INCLUDE_DIR}
	${PROJECT_BINARY_DIR}
	${OPENGL_INCLUDE_DIRS}
	${GLFW_INCLUDE_DIRS}
	${GLEW_INCLUDE_DIRS})

add_executable(${PROJECT_NAME} ${SOURCES})
set_target_properties(${PROJECT_NAME} PROPERTIES MACOSX_BUNDLE TRUE)
add_definitions(
	-std=c++11)

target_link_libraries(${PROJECT_NAME}
	${OPENGL_LIBRARIES}
	${GLFW_LIBRARIES}
	${GLEW_LIBRARIES})
